def get_dls(df, tok=None, bs=128):
    db = DataBlock(
        blocks=(TextBlock.from_df("text", tok=tok, seq_len=72), CategoryBlock),
        get_x=ColReader("text"),
        get_y=ColReader("label"),
        splitter=RandomSplitter(0.2)
    )
    return db.dataloaders(df, bs=bs)
